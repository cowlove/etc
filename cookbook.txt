
# ip forwarding

/usr/sbin/iptables -t nat -A POSTROUTING -o wlan1 -j MASQUERADE
/usr/sbin/iptables -A FORWARD -i wlan1 -o wlan0 -m state --state RELATED,ESTABLISHED -j ACCEPT
